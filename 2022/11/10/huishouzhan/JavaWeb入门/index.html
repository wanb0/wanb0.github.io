5.编写dao类

  • BaseDAO

    package com.dao;


import org.testng.annotations.Test;
import java.util.Collections;
import java.lang.reflect.Field;
import java.sql.*;
import java.util.ArrayList;
import java.util.List;

import static com.alibaba.druid.util.JdbcUtils.close;


/**
 *针对数据表的通用的操作
 */
public abstract class BaseDAO {

    //通用的增删改操作
    @Test
    public int UpDate(Connection conn, String sql, Object... args) throws SQLException {
        PreparedStatement pstmt = null;
        try {
            //预编译sql语句
            pstmt = conn.prepareStatement(sql);
            //填充占位符
            for (int i = 0; i < args.length; i++) {
                pstmt.setObject(i + 1, args[i]);
            }
            //执行sql语句
            return pstmt.executeUpdate();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            //关闭资源
            close(pstmt);
        }
        return 0;
    }
    //通用的查询操作，返回表中的一条记录
    @Test
    public <T> T getInstance(Connection conn, Class<T> clazz, String sql, Object... args) throws SQLException {
        PreparedStatement pstmt = null;
        ResultSet rs = null;
        try {
            //预编译sql语句
            pstmt = conn.prepareStatement(sql);
            //填充占位符
            for (int i = 0; i < args.length; i++) {
                pstmt.setObject(i + 1, args[i]);
            }
            //执行sql语句
            rs = pstmt.executeQuery();
            //获取结果集的元数据
            ResultSetMetaData rsmd = rs.getMetaData();
            //获取列数
            int columnCount = rsmd.getColumnCount();
            if (rs.next()) {
                T t = clazz.newInstance();
                //处理结果集一行数据中的每一列
                for (int i = 0; i < columnCount; i++) {
                    //获取列值
                    Object columnValue = rs.getObject(i + 1);
                    //获取每个列的列名
                    String columnLabel = rsmd.getColumnLabel(i + 1);
                    //给t对象指定的columnName属性，赋值为columnValue，通过反射
                    Field field = clazz.getDeclaredField(columnLabel);
                    field.setAccessible(true);
                    field.set(t, columnValue);
                }
                return t;
            }
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            close(rs);
            close(pstmt);
        }
        return null;
    }
    //通用的查询操作，返回表中的多条记录构成的集合
    @Test
    public <T> List<T> getForList(Connection conn, Class<T> clazz, String sql, Object... args) throws SQLException {
        PreparedStatement pstmt = null;
        ResultSet rs = null;
        try {
            //预编译sql语句
            pstmt = conn.prepareStatement(sql);
            //填充占位符
            for (int i = 0; i < args.length; i++) {
                pstmt.setObject(i + 1, args[i]);
            }
            //执行sql语句
            rs = pstmt.executeQuery();
            //获取结果集的元数据
            ResultSetMetaData rsmd = rs.getMetaData();
            //获取列数
            int columnCount = rsmd.getColumnCount();
            //创建集合对象
            ArrayList<T> list = new ArrayList<>();
            while (rs.next()) {
                T t = clazz.newInstance();
                //处理结果集一行数据中的每一列
                for (int i = 0; i < columnCount; i++) {
                    //获取列值
                    Object columnValue = rs.getObject(i + 1);
                    //获取每个列的列名
                    String columnLabel = rsmd.getColumnLabel(i + 1);
                    //给t对象指定的columnName属性，赋值为columnValue，通过反射
                    Field field = clazz.getDeclaredField(columnLabel);
                    field.setAccessible(true);
                    field.set(t, columnValue);
                }
                list.add(t);
            }
            return list;
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            close(rs);
            close(pstmt);
        }
        return null;
    }
    //用于查询特殊值的通用的方法
    @Test
    public <E> E getValue(Connection conn, String sql, Object... args) {
        PreparedStatement pstmt = null;
        ResultSet rs = null;
        try {
            //预编译sql语句
            pstmt = conn.prepareStatement(sql);
            //填充占位符
            for (int i = 0; i < args.length; i++) {
                pstmt.setObject(i + 1, args[i]);
            }
            //执行sql语句
            rs = pstmt.executeQuery();
            if (rs.next()) {
                return (E) rs.getObject(1);
            }
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            close(rs);
            close(pstmt);
        }
        return null;
    }


}

  • 编写FruitDAO

    package com.dao;

import com.fruitpojo.Fruit;

import java.sql.Connection;
import java.sql.Date;
import java.sql.SQLException;
import java.util.List;

/**
 *用来定义Fruit类的的规范操作
 */
public interface FruitDAO {
    /**
     * 将fruit对象添加到数据库中
     * @param fruit
     */
    void insert(Connection conn,Fruit fruit) throws SQLException;

    /**
     * 针对指定的id，删除表中的一条记录
     * @param id
     */
    void deleteById(Connection conn,int id) throws SQLException;

    /**
     * 针对内存中的fruit对象，去修改数据表中指定的记录
     * @param fruit
     */
    void update(Connection conn,Fruit fruit) throws SQLException;

    /**
     * 针对指定的id查询得到对应的Fruit对象
     * @param id
     */
    Fruit getFruitById(Connection conn,int id) throws SQLException;

    /**
     * 查询表中的所有记录构成的集合
     * @return
     */
    List<Fruit> getAll(Connection conn) throws SQLException;

    /**
     * 返回数据表中的数据的条目数
     * @return
     */
    Long getCount(Connection conn);
}

  • 编写FruitDAOImpl

    package com.dao;
import com.fruitpojo.Fruit;
import java.sql.Connection;
import java.sql.SQLException;
import java.util.List;

public class FruitDAOImpl extends BaseDAO implements FruitDAO {
    @Override
    public void insert(Connection conn, Fruit fruit) throws SQLException {
        String sql = "insert into fruit(fid,fname,price,fcount,remark) values(?,?,?,?,?)";
        UpDate(conn, sql, fruit.getFid(), fruit.getFname(), fruit.getPrice(), fruit.getFcount(), fruit.getRemark());
    }

    @Override
    public void deleteById(Connection conn, int id) throws SQLException {
        String sql = "delete from fruit where fid = ?";
        UpDate(conn, sql, id);
    }

    @Override
    public void update(Connection conn, Fruit fruit) throws SQLException {
        String sql = "update fruit set fname = ?,price = ?,fcount = ?,remark = ? where fid = ?";
        UpDate(conn, sql, fruit.getFname(), fruit.getPrice(), fruit.getFcount(), fruit.getRemark(), fruit.getFid());
    }

    @Override
    public Fruit getFruitById(Connection conn, int id) throws SQLException {
        String sql = "select fid,fname,price,fcount,remark from fruit where fid = ?";
        Fruit fruit = getInstance(conn, Fruit.class, sql, id);
        return fruit;
    }

    @Override
    public List<Fruit> getAll(Connection conn) throws SQLException {
        String sql = "select fid,fname,price,fcount,remark from fruit";
        List<Fruit> list = getForList(conn, Fruit.class, sql);
        return list;
    }

    @Override
    public Long getCount(Connection conn) {
        String sql = "select count(*) from fruit";
        Long count = getValue(conn, sql);
        return count;
    }
}
