1. 在idea中使用新建JavaWeb项目

  • 先打开一个项目，或者新建一个项目，在项目中新建一个Module

  • 新建一个测试页面，并把它部署到Tomcat中

    • 打开文件构建页面

    • 给新模块新建一个artifacts

    • 打开run的设置

    • 添加Tomcat的本地服务

    • 在deployment添加新增的artifacts

    • 在server中设置启动项

    • 在项目的web下新增一个web页面测试

    • 点击debug或者运行尝试是否启动成功

    • 显示刚才的web测试页面，即为部署成功

3. 新建一个Servlet类，用来获取表单数据

import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

public class AddServlets extends HttpServlet {
    @Override
    public void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        String fname = req.getParameter("fname");
        String priceStr = req.getParameter("price");
        Integer price = Integer.parseInt(priceStr);
        String fcount = req.getParameter("fcount");
        Integer count = Integer.parseInt(fcount);
        String remark = req.getParameter("remark");
        System.out.println(fname + price + count + remark);
    }

}
